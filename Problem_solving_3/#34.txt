
bool CheckIfCharIsVowel(char ToCheck) {
	ToCheck = tolower(ToCheck);
	return (ToCheck == 'a' || ToCheck == 'i' || ToCheck == 'e' || ToCheck == 'o' || ToCheck == 'u');
}

void PrintAlLVowelsInText(string Text) {
	
	short Counter = 0;
	for (short b = 0; b < Text.length(); b++) {
		if (CheckIfCharIsVowel(Text[b]))
		{
			cout << Text[b]<<" ";
	    }
	}

}
