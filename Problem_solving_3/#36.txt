short CountEachwordInText(string Text) {
	short Counter = 0;
	string Delim = " ";
	short Postion = 0;
	string Word = "";
	while ((Postion = Text.find(Delim)) != std::string::npos) {
		Word = Text.substr(0, Postion);
		if (Word != "") {
			Counter++;
		}
		Text.erase(0, Postion + Delim.length());
	}
	if (Text != "") {
		Counter++;
	}
	return Counter;
}



int main() {

	cout<<"The Number of word in your string is "<<CountEachwordInText(MyLib::ReadText());

	cout << "\n";
}
