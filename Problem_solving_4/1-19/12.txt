short CountFromTheBgOfYear(short Year,short month,short Day) {
	short NumberofDays = 0;
	for (short i = 1; i <= month - 1;i++) {
		NumberofDays += MyLib::NumberOfDaysInMonth(Year, i);

	}
	NumberofDays += Day;
	return NumberofDays;
}


stDay AddDaysToADate(short NumberOfDys,short Year,short Month,short Days) {
	stDay Day;
	Day.Year = Year;
	Day.Month = Month;
	short RemainingDays = NumberOfDys + Days;
	short GetDays = 0;
	while (true) {
		GetDays = MyLib::NumberOfDaysInMonth(Day.Year, Day.Month);
		if (RemainingDays > GetDays) {
			RemainingDays -= GetDays;
			Day.Month++;
			if (Day.Month == 1 && Month != 1) {
				Day.Year++;
			}
		}
		else {
			Day.Day = RemainingDays;
			break;
		}


	}

	return Day;

}
