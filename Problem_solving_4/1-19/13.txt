short CountFromTheBgOfYear(short Year,short month,short Day) {
	short NumberofDays = 0;
	for (short i = 1; i <= month - 1;i++) {
		NumberofDays += MyLib::NumberOfDaysInMonth(Year, i);

	}
	NumberofDays += Day;
	return NumberofDays;
}


bool CompareTwoDates(stDay Date1,stDay Date2) {
	return(CountFromTheBgOfYear(Date1.Year, Date1.Month, Date1.Day) < CountFromTheBgOfYear(Date2.Year, Date2.Month, Date2.Day) && Date1.Year <= Date2.Year);
}
