struct stDay {
	short Year;
	short Month;
	short Day;
};

bool CheckIfLastDayInMonth(stDay Date) {
	short GetDaysInMonth = MyLib::NumberOfDaysInMonth(Date.Year, Date.Month);
	return (GetDaysInMonth == Date.Day ? true : false);
}

bool CheckIfLastMonthInYear(short Month) {
	return (Month == 12 ? true : false);
}

bool AreDatesEqual(stDay Date1, stDay Date2) {
	return ((Date1.Year == Date2.Year) ? ((Date1.Month == Date2.Month) ? ((Date1.Day == Date2.Day) ? true : false) : false) : false);
}

stDay GetFullDate() {
	stDay Date;
	Date.Day = Getday();
	Date.Month = GetMonth();
	Date.Year = GetYear();

	return Date;
}

bool IsDate1BeforeDate2( stDay Date1 , stDay Date2) {
	return ((Date1.Year < Date2.Year) ? true : (Date1.Year == Date2.Year) ? (Date1.Month < Date2.Month) ? true : (Date1.Month == Date2.Month ? Date1.Day < Date2.Day : false) : false);

}


stDay IncressDateByOneDay(stDay Date) {
	if (CheckIfLastDayInMonth(Date)) {
		if (CheckIfLastMonthInYear(Date.Month)) {
			Date.Day = 1;
			Date.Month = 1;
			Date.Year++;
		}
		else {
			Date.Day = 1;
			Date.Month++;
		}

	}
	else {
		Date.Day++;
	}
	return Date;
}


int CountFromBirthDateTillNow(stDay Date1)
{
	stDay Date2;
	Date2.Year = 2026;
	Date2.Month = 1;
	Date2.Day = 20;
	int DaysFromPastTillNow = 0;
	while (IsDate1BeforeDate2(Date1,Date2)) {
		DaysFromPastTillNow++;
		Date1 = IncressDateByOneDay(Date1);
	}
	return DaysFromPastTillNow;
}
