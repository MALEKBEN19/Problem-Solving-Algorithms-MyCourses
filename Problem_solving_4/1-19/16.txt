void IncressDateByOneDay(stDay &Date) {
	short GetAllDaysInMonth = Lib2::NumberOfDaysInMonth(Date.Year, Date.Month);
	if (Date.Day != GetAllDaysInMonth) {
		Date.Day++;
	}
	else {
		Date.Day = 1;
		Date.Month++;
		if (Date.Month > 12) {
			Date.Month = 1;
			Date.Year++;
		}
	}
}
