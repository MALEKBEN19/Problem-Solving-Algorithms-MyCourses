bool AreDatesEqual(stDay Date1,stDay Date2) {
	return ((Date1.Year == Date2.Year) && (Date1.Month == Date2.Month) && (Date1.Day == Date2.Day));
}



int main() {

	cout << "Date 1 : \n";
	stDay Date1= GetFullDate();
	cout << "\n\n";
	cout << "Date 2 : \n";
	stDay Date2= GetFullDate();

	if (AreDatesEqual(Date1, Date2)) {
		cout << "Yes, Date 1 is Equal With Date 2.\n";
	}
	else {
		cout << "No, Date 1 is Not Equal With Date 2.\n";
	}

	system("pause>0");
	return 0;
}
