short CountFromTheBgOfYear(short Year,short month,short Day) {
	int counter = 0;
	string DateToString = "";
	string DateByUser= to_string(Year) + "." + to_string(month) + "." + to_string(Day);
	for (int M=1;M!=12;M++) {
		for (int d = 1; d <= MyLib::NumberOfDaysInMonth(Year, M); d++) {
			DateToString = to_string(Year) + "." + to_string(M) + "." + to_string(d);
			counter++;
			if (DateToString == DateByUser) {
				return counter;
			}
			DateToString = "";
		}
	}


}
