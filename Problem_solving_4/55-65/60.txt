bool IsDateWithinPeriod(stPeriod Period, stDate DateToCheck) {
	while (IsDate1BeforeDate2(Period.DateStart, Period.DateEnd)) {
		if (IsDate1EqualDate2(Period.DateStart, DateToCheck)) {
			return true;
		}
		Period.DateStart = IncressDateByOneDay(Period.DateStart);
	}
	return false;
}
