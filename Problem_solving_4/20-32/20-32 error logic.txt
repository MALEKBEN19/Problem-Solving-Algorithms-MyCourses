bool AreDatesEqual(stDay Date1, stDay Date2) {
	return ((Date1.Year == Date2.Year) ? ((Date1.Month == Date2.Month) ? ((Date1.Day == Date2.Day) ? true : false) : false) : false);
}

stDay GetFullDate() {
	stDay Date;
	Date.Day = Getday();
	Date.Month = GetMonth();
	Date.Year = GetYear();

	return Date;
}

bool IsDate1BeforeDate2( stDay Date1 , stDay Date2) {
	return ((Date1.Year < Date2.Year) ? true : (Date1.Year == Date2.Year) ? (Date1.Month < Date2.Month) ? true : (Date1.Month == Date2.Month ? Date1.Day < Date2.Day : false) : false);

}


stDay IncressDateByOneDay(stDay Date) {
	if (CheckIfLastDayInMonth(Date)) {
		if (CheckIfLastMonthInYear(Date.Month)) {
			Date.Day = 1;
			Date.Month = 1;
			Date.Year++;
		}
		else {
			Date.Day = 1;
			Date.Month++;
		}

	}
	else {
		Date.Day++;
	}
	return Date;
}

void SwapDates(stDay &Date1, stDay &Date2) {
	stDay tempDate;
	tempDate.Year = Date1.Year;
	tempDate.Month = Date1.Month;
	tempDate.Day = Date1.Day;

	Date1.Year = Date2.Year;
	Date1.Month = Date2.Month;
	Date1.Day = Date2.Day;

	Date2.Year = tempDate.Year;
	Date2.Month = tempDate.Month;
	Date2.Day = tempDate.Day;


}


int CountDifferencesBetweenTwoDates(stDay Date1, stDay Date2, bool IncludeLastDay = false)
{
	int DaysDiff = 0;
	short SwapFlagValue = 1;
	if (!IsDate1BeforeDate2(Date1, Date2)) {
		SwapDates(Date1, Date2);
		SwapFlagValue = -1;
	}

	while (IsDate1BeforeDate2(Date1, Date2)) {
		DaysDiff++;
		Date1 = IncressDateByOneDay(Date1);
	}
	return IncludeLastDay ? ++DaysDiff*SwapFlagValue : DaysDiff * SwapFlagValue;
}



string ConvertstructDateToString(stDay date) {
	string Datetostr = to_string(date.Day) + "/" + to_string(date.Month) + "/" + to_string(date.Year);
	return Datetostr;
}
