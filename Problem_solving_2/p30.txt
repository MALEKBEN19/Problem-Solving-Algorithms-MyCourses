#include <iostream>
using namespace std;
#include <string>
#include <cmath>
#include<cstdlib>


int RandomNumber(int from, int to) {
	int Randu = rand() % (to - from + 1) + from;
	return Randu;
}

void FillArryWithRandomNumbers(int arr[100],int &ArryLenght) {
	cout << "How many element you want?\n";
	cin >> ArryLenght;
	for (int x = 0; x < ArryLenght; x++) {
		arr[x] = RandomNumber(1,100);
	}
}

void PrintArry(int arr[100], int ArryLenght) {
	for (int f = 0; f < ArryLenght; f++) {
		cout << arr[f]<<" ";
	}
	cout << endl;
}
void SumOfTwoArries(int arr1[100],int arr2[100],int lenght) {

	for (int x = 0; x < lenght; x++) {
		cout << arr1[x] + arr2[x] << " ";
	}

	cout << endl;
}



int main(){
	srand((unsigned)time(NULL));
	int arr1[100],arr2[100], ArryLenght = 0;

	FillArryWithRandomNumbers(arr1,ArryLenght);
	

	cout << "Arry 1 element :\n";
	PrintArry(arr1,ArryLenght);
	cout << endl;
	FillArryWithRandomNumbers(arr2, ArryLenght);
	cout << "Arry 2 element :\n";
	PrintArry(arr2,ArryLenght);
	cout << endl;
	cout << "The sum of arry 1 And arry 2 is :\n";
	SumOfTwoArries(arr1,arr2,ArryLenght);

	return 0;
}
