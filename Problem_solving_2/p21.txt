int readPositiveNumber(string message) {
	int num = 0;
	do {
		cout << message << endl;
		cin >> num;
	} while (num <= 0);
	return num;
}

int RandomNumber(int from,int to) {
	int RandomN = rand()%(to-from+1)+from;
	return RandomN;
}
string GetFourLetter(){
	string FourLetter = "";
	char Firsletter = RandomNumber(65,90);
	char Secondtletter = RandomNumber(65, 90);
	char Thirdletter = RandomNumber(65, 90);
	char Fourthletter = RandomNumber(65, 90);
	FourLetter = FourLetter + Firsletter;
	FourLetter += Secondtletter;
	FourLetter += Thirdletter;
	FourLetter += Fourthletter;
	return FourLetter;
}
