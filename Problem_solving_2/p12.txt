int readPositiveNumber(string message) {
	int num = 0;
	do {
		cout << message << endl;
		cin >> num;
	} while (num <= 0);
	return num;
}

void InvertedNumber(int num){

	for (int x = 1; num>=x; num--) {
		for (int g = 1; g <= num; g++) {
			cout << num;
		}
		cout << endl;
	}
}


int main() {

	InvertedNumber(readPositiveNumber("Please enter positive number!"));


	return 0;
}
