int RandomNumber(int from, int to) {
	int RandomN = rand() % (to - from + 1) + from;
	return RandomN;
}

int readPositiveNumber(string message) {
	int num = 0;
	do {
		cout << message << endl;
		cin >> num;
	} while (num <= 0);
	return num;
}

}

void PrintArry(int arr[100], int length) {
	for (int x = 0; x<length; x++) {
		cout<<arr[x]<<" ";
	}
	cout << endl;
}
short CheckIfNumberInsideAnArry(int arr[100], int length,int searchfor) {
	
	int counter = 0;
	for (int h = 0; h < length; h++) {
		if (arr[h] == searchfor) {
			return counter;
		}
		counter++;
	}
	return -1;
}
