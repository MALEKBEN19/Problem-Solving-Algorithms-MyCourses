enum CheckOddOrEven {odd=1,even=2};
int RandomNumber(int from, int to) {
	int RandomN = rand() % (to - from + 1) + from;
	return RandomN;
}
CheckOddOrEven CheckOdd(int number) {
	if (number % 2 != 0)
		return CheckOddOrEven::odd;
	else {
		return CheckOddOrEven::even;
	}
}
