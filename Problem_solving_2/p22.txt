int ReadPositiveNum(string Message) {
	int num = 0;
	do {
		cout << Message << endl;
		cin >> num;
	} while (num<=0);
	return num;
}
void FillArry(int arr[100],int lenght) {
	int counter = 0;
	for (int i = 0; i < lenght;i+=1) {
		counter++;
		cout << "Element[" << counter << "] : ";
		cin >> arr[i];
	}
	cout << endl;
}

string fromArrytostring(int arr[100],int lenght) {
	
	string fullValues = "";
	for (int x = 0; x < lenght; x++) {
		 fullValues=fullValues+ to_string(arr[x]);
	}
	return fullValues;
}

	int fullValue = stoi(fullvalue);
	int Remiander = 0;

		Remiander = fullValue % 10;
		fullValue = fullValue / 10;
		num1 = num1 * 10 + Remiander;
	}

}
int fullValueInInteger(int reversedValue) {

		Remiander = reversedValue % 10;

}
void HowManyNumberTocheckRepeatedInArry(int arr[100], int NumberToCheck,int length){

		}

}
